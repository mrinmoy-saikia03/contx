# contx — Plan 5: Local Read-only Web UI

> **For agentic workers:** REQUIRED SUB-SKILL: superpowers:subagent-driven-development.

**Goal:** `contx serve` launches a local web server that lets humans browse the repo's intent map. Read-only. No auth. Single localhost port. Server-rendered HTML via Jinja2 templates + minimal htmx for search interactivity. No SPA, no build step.

**Architecture:** New `contx/web/` package with a FastAPI app. FastAPI is already an indirect dependency of `mcp`, so the addition cost is small (jinja2, uvicorn-as-runner). The app reuses `contx.store.read_entries` + `contx.store.fold_entries` + `contx.search.search_entries` directly — no new business logic.

**Tech Stack:** Python 3.11+, FastAPI, Jinja2, uvicorn, htmx (loaded from a CDN — no bundling).

```python
"""Read-only web UI for browsing contx entries in a repo."""

from __future__ import annotations

import os
from pathlib import Path

from fastapi import FastAPI, Request
from fastapi.responses import HTMLResponse
from fastapi.templating import Jinja2Templates

from contx.repo import find_repo_root

_TEMPLATES_DIR = Path(__file__).parent / "templates"


def _repo_root() -> Path:
    override = os.environ.get("CONTX_REPO_ROOT")
    return find_repo_root(Path(override) if override else Path.cwd())


def create_app(repo_root: Path | None = None) -> FastAPI:
    """Construct the FastAPI app. Pass repo_root in tests; defaults to env/cwd."""
    app = FastAPI(title="contx", description="Read-only viewer for contx intent")
    templates = Jinja2Templates(directory=str(_TEMPLATES_DIR))
    root = repo_root or _repo_root()

    @app.get("/", response_class=HTMLResponse)
    def index(request: Request) -> HTMLResponse:
        files = _list_tracked_files(root)
        return templates.TemplateResponse(
            request, "index.html",
            {"files": files, "repo": str(root)},
        )

    return app


def _list_tracked_files(repo_root: Path) -> list[str]:
    """Walk .contx/ and return source-relative paths for all sidecars."""
    from contx.paths import CTX_DIR, SIDECAR_SUFFIX, source_path_for_sidecar
    ctx_dir = repo_root / CTX_DIR
    if not ctx_dir.is_dir():
        return []
    out: list[str] = []
    for sidecar in sorted(ctx_dir.rglob(f"*{SIDECAR_SUFFIX}")):
        try:
            src = source_path_for_sidecar(repo_root, sidecar)
            out.append(str(src.relative_to(repo_root)))
        except ValueError:
            continue
    return out
```

```python
    @app.get("/file/{file_path:path}", response_class=HTMLResponse)
    def file_view(file_path: str, request: Request) -> HTMLResponse:
        from fastapi import HTTPException
        from contx.store import fold_entries, read_entries
        entries = read_entries(root, file_path)
        if not entries:
            raise HTTPException(status_code=404, detail=f"No context for {file_path}")
        folded = fold_entries(entries)
        return templates.TemplateResponse(
            request, "file.html",
            {"file": file_path, "folded": folded, "log": [_entry_view(e) for e in entries]},
        )

    @app.get("/symbol/{ref:path}", response_class=HTMLResponse)
    def symbol_view(ref: str, request: Request) -> HTMLResponse:
        from fastapi import HTTPException
        from contx.paths import parse_symbol_ref
        from contx.store import fold_entries, read_entries
        try:
            file_path, symbol = parse_symbol_ref(ref)
        except ValueError as exc:
            raise HTTPException(status_code=400, detail=str(exc))
        if symbol is None:
            raise HTTPException(status_code=400, detail="symbol path required (file::symbol)")
        entries = read_entries(root, file_path)
        symbol_entries = [e for e in entries if e.symbol == symbol]
        if not symbol_entries:
            raise HTTPException(status_code=404, detail=f"No context for {ref}")
        folded = fold_entries(entries)
        return templates.TemplateResponse(
            request, "symbol.html",
            {
                "file": file_path,
                "symbol": symbol,
                "intent": folded.symbols.get(symbol),
                "log": [_entry_view(e) for e in symbol_entries],
            },
        )
```

```python
    @app.get("/search", response_class=HTMLResponse)
    def search_view(request: Request, q: str | None = None) -> HTMLResponse:
        from contx.search import search_entries
        hits = search_entries(root, q, limit=100) if q else []
        return templates.TemplateResponse(
            request, "search.html",
            {"q": q or "", "hits": hits},
        )

    @app.get("/timeline", response_class=HTMLResponse)
    def timeline_view(request: Request, limit: int = 50) -> HTMLResponse:
        from contx.paths import CTX_DIR, SIDECAR_SUFFIX, source_path_for_sidecar
        from contx.store import read_entries

        items: list[dict] = []
        ctx_dir = root / CTX_DIR
        if ctx_dir.is_dir():
            for sidecar in ctx_dir.rglob(f"*{SIDECAR_SUFFIX}"):
                try:
                    src = source_path_for_sidecar(root, sidecar)
                except ValueError:
                    continue
                rel = str(src.relative_to(root))
                for e in read_entries(root, rel):
                    items.append({"file": rel, **_entry_view(e)})
        items.sort(key=lambda i: i["timestamp"], reverse=True)
        items = items[:limit]
        return templates.TemplateResponse(
            request, "timeline.html",
            {"items": items},
        )

@app.command()
def serve(
    port: int = typer.Option(4242, "--port", "-p", help="Port to bind (default 4242)"),
    host: str = typer.Option("127.0.0.1", "--host", help="Bind host (default 127.0.0.1)"),
) -> None:
    """Launch the local read-only web UI."""
    import uvicorn
    from contx.web.app import create_app
    repo = _resolve_repo()
    if not is_initialized(repo):
        typer.echo("error: contx not initialized for this repo. Run `contx init` first.", err=True)
        raise typer.Exit(code=2)
    web_app = create_app(repo_root=repo)
    typer.echo(f"contx serving on http://{host}:{port}")
    uvicorn.run(web_app, host=host, port=port, log_level="warning")
```
